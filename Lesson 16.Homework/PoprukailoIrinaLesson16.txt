3. Найти больных, которые повторно обратились в больницу c тем же диагнозом 
в течение одного года. Вывести все посещения, а для них: больного, диагноз, 
дату, врача.

SELECT п.ID_Пациента, 
       п.ФИО, 
       д.Название_заболевания, 
       пб.Дата_обращения,
       в.ФИО
FROM Пациенты п
JOIN 
    Прием_больных пб ON пб.ID_Пациента=п.ID_Пациента
JOIN 
     Диагнозы д ON д.ID_диагноза=пб.ID_диагноза
JOIN 
     Врачи в ON в.ID_врача=пб.ID_врача
JOIN 
     Прием_больных пб2 ON пб2.ID_Пациента=пб.ID_Пациента
                       AND пб2.ID_диагноза=пб.ID_диагноза
                       AND пб2.Дата_обращения>пб.Дата_обращения
WHERE DATEDIFF(DAY, пб.дата_обращения, пб2.дата_обращения) <= 365
ORDER BY п.ID_Пациента, п.ФИО, д.Название_заболевания,пб.Дата_обращения, в.ФИО
